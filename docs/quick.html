Result Types

SCULPTOR_RESULT (XOR Pattern)

if l_result.is_success then
    mesh := l_result.mesh
    -- Process 3D mesh
else
    error := l_result.error_message
    -- Handle error
end

SCULPTOR_INFERENCE_RESULT

if l_result.is_success then
    points := l_result.points  -- POINT_CLOUD
else
    error := l_result.error_message
end

MESH_VALIDATION_REPORT

if l_report.is_valid then
    -- Mesh topology is valid
else
    errors := l_report.error_messages  -- LIST [STRING]
    count := l_report.error_count
end

Common Patterns

Builder Pattern

l_sculptor.set_device ("CPU")
    .set_voxel_size (0.5)
    .set_seed (42)
    .set_num_inference_steps (64)

Error Handling (XOR Result)

result := l_sculptor.generate ("cube")
if result.is_success and attached result.mesh as m then
    -- Process m
else if attached result.error_message as err then
    print (err)
end

Batch Processing

prompts := ["red cube", "blue sphere", "green pyramid"]
results := l_sculptor.batch_generate (prompts)
across results as r loop
    if r.item.is_success then
        process (r.item.mesh)
    end
end
